                    <input id="bookTitle" class="form-control mb-2" placeholder="Book Title">
                    <input id="bookAuthor" class="form-control mb-2" placeholder="Author">
                    <input id="bookCopies" type="number" class="form-control mb-2" placeholder="Copies" value="1">
                    <button onclick="saveBook()" class="btn btn-primary w-100">
                        <i class="bi bi-save"></i> Save Book
                    </button>

            </div>
        </div>

        <!-- MEMBER FORM -->
        <div class="col-md-6">
            <div class="card shadow-sm">

                    <i class="bi bi-person-plus"></i> Add / Edit Member
                </div>
                <div class="card-body">
                    <input id="memberId" class="form-control mb-2" placeholder="Member ID">
                    <input id="memberName" class="form-control mb-2" placeholder="Member Name">
                    <button onclick="saveMember()" class="btn btn-success w-100">
                        <i class="bi bi-save"></i> Save Member
                    </button>

        <!-- BORROW -->

                <div class="card-body">
                    <select id="borrowMember" class="form-select mb-2"></select>
                    <select id="borrowBookSelect" class="form-select mb-2"></select>
                    <button onclick="borrowBook()" class="btn btn-warning w-100">
                        <i class="bi bi-arrow-down"></i> Borrow
                    </button>
                </div>
            </div>
        </div>

        <!-- RETURN -->
        <div class="col-md-6">
            <div class="card shadow-sm">

                </div>
                <div class="card-body">
                    <select id="returnMember" class="form-select mb-2"></select>
                    <select id="returnBookSelect" class="form-select mb-2"></select>
                    <button onclick="returnBook()" class="btn btn-danger w-100">
                        <i class="bi bi-arrow-up"></i> Return
                    </button>

    <!-- HISTORY -->

<script>
    // Demo data with Indian names and books
    let books = [
        { id: "B1", title: "Wings of Fire", author: "A.P.J. Abdul Kalam", copies: 5, borrowed: 0 },
        { id: "B2", title: "The God of Small Things", author: "Arundhati Roy", copies: 4, borrowed: 0 },
        { id: "B3", title: "The White Tiger", author: "Aravind Adiga", copies: 6, borrowed: 0 },
        { id: "B4", title: "Train to Pakistan", author: "Khushwant Singh", copies: 3, borrowed: 0 },
        { id: "B5", title: "Gitanjali", author: "Rabindranath Tagore", copies: 7, borrowed: 0 },
        { id: "B6", title: "The Palace of Illusions", author: "Chitra Banerjee Divakaruni", copies: 5, borrowed: 0 },
        { id: "B7", title: "Immortals of Meluha", author: "Amish Tripathi", copies: 8, borrowed: 0 },
        { id: "B8", title: "Five Point Someone", author: "Chetan Bhagat", copies: 10, borrowed: 0 }
    ];

    let members = [
        { id: "M1", name: "Aarav Sharma", borrowedBooks: [], lateFees: 0 },
        { id: "M2", name: "Priya Patel", borrowedBooks: [], lateFees: 0 },
        { id: "M3", name: "Rohan Mehta", borrowedBooks: [], lateFees: 0 },
        { id: "M4", name: "Ananya Iyer", borrowedBooks: [], lateFees: 0 },
        { id: "M5", name: "Vikram Singh", borrowedBooks: [], lateFees: 0 },
        { id: "M6", name: "Sanya Kapoor", borrowedBooks: [], lateFees: 0 },
        { id: "M7", name: "Arjun Nair", borrowedBooks: [], lateFees: 0 },
        { id: "M8", name: "Meera Joshi", borrowedBooks: [], lateFees: 0 }
    ];

    let history = [];

    let editBookIndex = -1;
    let editMemberIndex = -1;

    const LOAN_DAYS = 7;
    const LATE_FEE_PER_DAY = 1;

    const borrowMemberSelect = document.getElementById("borrowMember");
    const borrowBookSelect = document.getElementById("borrowBookSelect");
    const returnMemberSelect = document.getElementById("returnMember");
    const returnBookSelect = document.getElementById("returnBookSelect");

    function saveBook() {
        const id = bookId.value.trim();
        const title = bookTitle.value.trim();
        const author = bookAuthor.value.trim();
        const copies = parseInt(bookCopies.value);

        if (!id || !title || !author || isNaN(copies) || copies < 1) {
            alert("Please fill all book fields correctly.");
            return;
        }

        if (editBookIndex === -1) {
            books.push({ id, title, author, copies, borrowed: 0 });
        } else {
            const existingBorrowed = books[editBookIndex].borrowed;
            books[editBookIndex] = { id, title, author, copies, borrowed: existingBorrowed };
            editBookIndex = -1;
        }

        clearBookForm();
        updateUI();
    }

    function saveMember() {
        const id = memberId.value.trim();
        const name = memberName.value.trim();

        if (!id || !name) {
            alert("Please fill all member fields.");
            return;
        }

        if (editMemberIndex === -1) {
            members.push({ id, name, borrowedBooks: [], lateFees: 0 });
        } else {
            members[editMemberIndex].id = id;
            members[editMemberIndex].name = name;
            editMemberIndex = -1;
        }

        clearMemberForm();
        updateUI();
    }

    function editBook(index) {
        const b = books[index];
        bookId.value = b.id;
        bookTitle.value = b.title;
        bookAuthor.value = b.author;
        bookCopies.value = b.copies;
        editBookIndex = index;
    }

    function editMember(index) {
        const m = members[index];
        memberId.value = m.id;
        memberName.value = m.name;
        editMemberIndex = index;
    }

    function deleteBook(index) {
        if (books[index].borrowed > 0) {
            alert("Cannot delete a book that is currently borrowed.");
            return;
        }
        if (confirm("Delete this book?")) {
            books.splice(index, 1);
            updateUI();
        }
    }

    function deleteMember(index) {
        if (members[index].borrowedBooks.length > 0) {
            alert("Cannot delete a member who has borrowed books.");
            return;
        }
        if (confirm("Delete this member?")) {
            members.splice(index, 1);
            updateUI();
        }
    }

    function clearBookForm() {
        bookId.value = "";
        bookTitle.value = "";
        bookAuthor.value = "";
        bookCopies.value = "1";
    }

    function clearMemberForm() {
        memberId.value = "";
        memberName.value = "";
    }

    function borrowBook() {
        const memberIdVal = borrowMemberSelect.value;
        const bookIdVal = borrowBookSelect.value;

        const member = members.find(m => m.id === memberIdVal);
        const book = books.find(b => b.id === bookIdVal);

        if (!member || !book) {
            alert("Select valid member and book.");
            return;
        }

        if (book.borrowed >= book.copies) {
            alert("No copies available.");
            return;
        }

        if (member.borrowedBooks.some(b => b.bookId === bookIdVal)) {
            alert("This member already borrowed this book.");
            return;
        }

        const borrowDate = new Date();
        const dueDate = new Date();
        dueDate.setDate(borrowDate.getDate() + LOAN_DAYS);

        book.borrowed++;
        member.borrowedBooks.push({ bookId: bookIdVal, borrowDate, dueDate });

        history.push({
            type: "BORROW",
            member: member.name,
            book: book.title,
            date: borrowDate.toLocaleString()
        });

        updateUI();
    }

    function returnBook() {
        const memberIdVal = returnMemberSelect.value;
        const bookIdVal = returnBookSelect.value;

        const member = members.find(m => m.id === memberIdVal);
        const book = books.find(b => b.id === bookIdVal);

        if (!member || !book) {
            alert("Select valid member and book.");
            return;
        }

        const record = member.borrowedBooks.find(b => b.bookId === bookIdVal);
        if (!record) {
            alert("This member did not borrow that book.");
            return;
        }

        const returnDate = new Date();
        const dueDate = new Date(record.dueDate);

        let lateFee = 0;
        if (returnDate > dueDate) {
            const lateDays = Math.ceil((returnDate - dueDate) / (1000 * 60 * 60 * 24));
            lateFee = lateDays * LATE_FEE_PER_DAY;
            member.lateFees += lateFee;
        }

        member.borrowedBooks = member.borrowedBooks.filter(b => b.bookId !== bookIdVal);
        book.borrowed--;

        history.push({
            type: "RETURN",
            member: member.name,
            book: book.title,
            date: returnDate.toLocaleString(),
            lateFee
        });

        updateUI();
    }

    function updateUI() {
        // BOOK TABLE
        bookList.innerHTML = `
            <table class="table table-striped">
                <thead class="table-dark">
                    <tr>
                        <th>ID</th><th>Title</th><th>Author</th><th>Available</th><th>Actions</th>
                    </tr>
                </thead>
                <tbody>
                    ${books.map((b, i) => `
                        <tr>
                            <td>${b.id}</td>
                            <td>${b.title}</td>
                            <td>${b.author}</td>
                            <td>${b.copies - b.borrowed} / ${b.copies}</td>
                            <td>
                                <button class="btn btn-sm btn-warning" onclick="editBook(${i})"><i class="bi bi-pencil"></i></button>
                                <button class="btn btn-sm btn-danger" onclick="deleteBook(${i})"><i class="bi bi-trash"></i></button>
                            </td>
                        </tr>
                    `).join("")}
                </tbody>
            </table>
        `;

        // MEMBER TABLE
        memberList.innerHTML = `
            <table class="table table-bordered">
                <thead class="table-secondary">
                    <tr>
                        <th>ID</th><th>Name</th><th>Borrowed Books</th><th>Late Fees</th><th>Actions</th>
                    </tr>
                </thead>
                <tbody>
                    ${members.map((m, i) => `
                        <tr>
                            <td>${m.id}</td>
                            <td>${m.name}</td>
                            <td>
                                ${m.borrowedBooks.map(b => {
                                    const book = books.find(x => x.id === b.bookId);
                                    const overdue = new Date() > new Date(b.dueDate);
                                    return `
                                        ${book ? book.title : b.bookId}
                                        <br><small>Due: ${new Date(b.dueDate).toLocaleDateString()}
                                        ${overdue ? "<span class='overdue'>(OVERDUE)</span>" : ""}
                                        </small><br>
                                    `;
                                }).join("") || "None"}
                            </td>
                            <td>₹${m.lateFees}</td>
                            <td>
                                <button class="btn btn-sm btn-warning" onclick="editMember(${i})"><i class="bi bi-pencil"></i></button>
                                <button class="btn btn-sm btn-danger" onclick="deleteMember(${i})"><i class="bi bi-trash"></i></button>
                            </td>
                        </tr>
                    `).join("")}
                </tbody>
            </table>
        `;

        // DROPDOWNS
        borrowMemberSelect.innerHTML = members.map(m => `<option value="${m.id}">${m.name}</option>`).join("");
        returnMemberSelect.innerHTML = borrowMemberSelect.innerHTML;

        borrowBookSelect.innerHTML = books.map(b => `<option value="${b.id}">${b.title}</option>`).join("");

        updateReturnBookOptions();
        updateHistory();
    }

    function updateReturnBookOptions() {
        const member = members.find(m => m.id === returnMemberSelect.value);
        if (!member) {
            returnBookSelect.innerHTML = "";
            return;
        }
        returnBookSelect.innerHTML = member.borrowedBooks
            .map(b => {
                const book = books.find(x => x.id === b.bookId);
                return `<option value="${b.bookId}">${book ? book.title : b.bookId}</option>`;
            })
            .join("");
    }

    returnMemberSelect.addEventListener("change", updateReturnBookOptions);

    function updateHistory() {
        historyList.innerHTML = history.map(h =>
            `<div class="border rounded p-2 mb-2">
                <strong>${h.type}</strong> — ${h.book} by ${h.member}
                <br>${h.date}
                ${h.lateFee ? `<br><span class="text-danger">Late Fee: ₹${h.lateFee}</span>` : ""}
            </div>`
        ).join("");
    }

    // Initial render
    updateUI();
</script>
